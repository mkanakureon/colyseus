## 1. ワールド概要

### 世界観

**アルカディア大陸** — かつて古代魔法文明が栄えた大陸。文明崩壊から数百年が経ち、人々は小さな村や街で暮らしている。最近、大陸各地で古代遺跡が活性化し、魔物が出現するようになった。冒険者たちは謎を解き明かすべく旅立つ。

### ワールド構成

全12ゾーンで構成。大きく4つのエリアに分類される。

| エリア | ゾーン数 | 特徴 |
|--------|---------|------|
| 町・村（安全地帯） | 4 | 戦闘なし。NPC・商店・クエスト受注 |
| フィールド（危険地帯） | 4 | 雑魚敵出現。探索・素材収集 |
| ダンジョン（高危険） | 3 | 強敵・ボス。パーティ推奨 |
| 特殊エリア | 1 | エンドコンテンツ・イベント用 |

## 7. TypeScript 型定義

```typescript
// src/zones/types.ts

// ==============================
// ゾーン定義
// ==============================

/** ゾーン間の隣接関係 */
interface AdjacentZone {
  direction: "north" | "south" | "east" | "west";
  zoneId: string;
  /** 通行条件（クエストクリア等）。undefined なら常時通行可 */
  requirement?: string;
}

/** ゾーンの種別 */
type ZoneType = "safe" | "field" | "dungeon" | "special";

/** ゾーン定義 */
interface ZoneDefinition {
  /** ゾーンID（例: "zone-01-village"） */
  id: string;
  /** 表示名 */
  name: string;
  /** ゾーンの説明テキスト */
  description: string;
  /** ゾーン種別 */
  type: ZoneType;
  /** 同時接続上限 */
  maxPlayers: number;
  /** 隣接ゾーン */
  adjacentZones: AdjacentZone[];
  /** 配置NPC一覧 */
  npcs: NPCDefinition[];
  /** 敵出現テーブル（field/dungeon のみ） */
  enemies?: EnemySpawnEntry[];
  /** 推奨レベル範囲 */
  recommendedLevel?: { min: number; max: number };
  /** 背景アセットID（kaedevn のアセットに対応） */
  bgAssetId?: string;
  /** 背景の説明（アセット未作成時のプレースホルダー） */
  bgDescription?: string;
  /** ゾーン固有イベント */
  events?: ZoneEvent[];
}

// ==============================
// NPC 定義
// ==============================

/** NPC定義 */
interface NPCDefinition {
  /** NPC ID（例: "npc-elder-yohan"） */
  id: string;
  /** 表示名 */
  name: string;
  /** 初期表情（kaedevn 18表情から選択） */
  expression: Expression;
  /** 初期ポーズ（kaedevn 16ポーズから選択） */
  pose: Pose;
  /** ゾーン内X座標（論理解像度 1280x720） */
  x: number;
  /** ゾーン内Y座標 */
  y: number;
  /** 固定セリフ一覧（インラインタグ [e:][p:][cam:][click] 付き） */
  dialogue: string[];
  /** NPC の役割 */
  role: "quest" | "shop" | "info" | "guard" | "boss" | "ambient";
  /** AI対話を有効にするか（コスト注意: 保留機能） */
  aiDialogue?: boolean;
}

/** kaedevn 表情（18種） */
type Expression =
  | "normal" | "smile" | "laugh" | "grin" | "wink" | "smirk"
  | "sad" | "cry" | "shy"
  | "angry" | "furious" | "serious"
  | "surprised" | "shocked" | "blush"
  | "sleepy" | "closed" | "cold";

/** kaedevn ポーズ（16種） */
type Pose =
  | "standing" | "relaxed" | "cool"
  | "arms_crossed" | "hands_on_hips" | "wave" | "point" | "salute" | "think"
  | "cheer" | "fighting" | "jump"
  | "bow" | "shy_pose" | "tired" | "sit";

// ==============================
// 敵定義
// ==============================

/** 敵のステータス定義 */
interface EnemyDefinition {
  /** 敵ID（例: "enemy-slime"） */
  id: string;
  /** 表示名 */
  name: string;
  /** HP */
  hp: number;
  /** 攻撃力 */
  attack: number;
  /** 防御力 */
  defense: number;
  /** 撃破時獲得経験値 */
  exp: number;
  /** 撃破時獲得ゴールド */
  gold: number;
  /** ドロップアイテム */
  drops: DropItem[];
  /** ボスフラグ */
  isBoss: boolean;
  /** 戦闘開始時の演出テキスト（インラインタグ付き） */
  encounterText?: string;
  /** 撃破時の演出テキスト（インラインタグ付き） */
  defeatText?: string;
}

/** ドロップアイテム */
interface DropItem {
  itemId: string;
  name: string;
  /** ドロップ率（0.0 - 1.0） */
  dropRate: number;
}

/** ゾーン内の敵出現エントリ */
interface EnemySpawnEntry {
  /** 敵定義への参照ID */
  enemyId: string;
  /** 出現重み（高いほど出やすい） */
  weight: number;
  /** 最大同時出現数 */
  maxSpawn: number;
  /** リスポーン間隔（秒） */
  respawnInterval: number;
}

// ==============================
// イベント定義
// ==============================

/** ゾーンイベント */
interface ZoneEvent {
  id: string;
  name: string;
  type: "quest" | "gimmick" | "collection" | "boss" | "timed" | "ambient";
  description: string;
  /** トリガー条件 */
  trigger: "interact" | "enter" | "timer" | "quest_progress";
  /** 関連NPC ID */
  npcId?: string;
  /** 必要クエスト（前提条件） */
  prerequisiteQuest?: string;
}
```
